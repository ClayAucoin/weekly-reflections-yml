    // World.frameRate = 5;
    //Create a bee sprite
    var bee = createSprite(200, 200);
    bee.setAnimation("bee");

    function draw(){
        background("lightblue");
        
        //Update the position of the bee to the position of the mouse
        if(World.mouseX < bee.x){
            bee.x = World.mouseX + randomNumber(-50, 50);
        }

        if(World.mouseX > bee.x){
            bee.x = World.mouseX + randomNumber(-50, 50);
        }

        if(World.mouseY < bee.y){
            bee.y = World.mouseY + randomNumber(-50, 50);
        }

        if(World.mouseY > bee.y){
            bee.y = World.mouseY + randomNumber(-50, 50);
        }

        drawSprites();
    }
